def tvd(dist1, dist2):
    return sum(abs(dist1 - dist2))/2

def simulated_tvd():
    return tvd(sample_proportions(1423, model), model)

num_simulations = 10000
for i in np.arange(num_simulations):
    new_tvd = simulated_tvd()
    tvds = np.append(tvds, new_tvd)
